public class ArrayTraversal {
    private int[] array;
    private int[][] matrix;

    public ArrayTraversal(int[][] matrix) {
        this.matrix = matrix;
    }

    public ArrayTraversal(int[] array) {
        this.array = array;
    }

    private void dialognal() {
        int[] result = findDiagonalOrder(this.matrix);
        System.out.println(Arrays.toString(result));
    }



    private int[] findDiagonalOrder(int[][] matrix) {
//        if the length of row and col is empty
        if (matrix.length == 0 || matrix[0].length == 0)
            return new int[0];
        int m = matrix.length, n = matrix[0].length;
        int i = 0;
        int[] arr = new int[m * n];
        int row = 0, col = 0;
        boolean up = true;
        while (row < m && col < n) {
            if (up) {
                while (row > 0 && col < n - 1) {
                    arr[i++] = matrix[row][col];
                    row--;
                    col++;
                }
                arr[i++] = matrix[row][col];
                if (col == n - 1) {
                    row++;
                } else {
                    col++;
                }
            } else {
                while (col > 0 && row < m - 1) {
                    arr[i++] = matrix[row][col];
                    row++;
                    col--;
                }
                arr[i++] = matrix[row][col];
                if (row == m - 1) {
                    col++;
                } else {
                    row++;
                }
            }
            if (up) up = false;
            else up = true;
        }

return  arr;
    }





    public static void main(String[] args) {
        ArrayTraversal a = new ArrayTraversal(new int[][]{{1, 2, 3}, {4, 5, 6}, {7, 8, 9}});
        a.dialognal();
    }

}
